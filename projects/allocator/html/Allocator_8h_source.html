110  bool valid () const {

112  for(const char* i = a; i < &a[N-sizeof(int)];)
113  {

115 
116  if (*(int*)i != *(int*)(i+diff)) //Does the sentinel have a matching
117  //sentinel at the appropriate address?
118  return false;
119 
120  diff += sizeof(int);
121  if(*(int*)i > 0) //If the block is free:
122  {
